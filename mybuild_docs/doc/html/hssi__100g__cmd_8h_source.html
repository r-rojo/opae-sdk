93  virtual int run(test_afu *afu, CLI::App *app) override
94  {
95  (void)app;
96 
97  hssi_afu *hafu = dynamic_cast<hssi_afu *>(afu);
98 
99  uint64_t bin_src_addr = mac_bits_for(src_addr_);
100  if (bin_src_addr == INVALID_MAC) {
101  std::cerr << "invalid MAC address: " << src_addr_ << std::endl;
102  return test_afu::error;
103  }
104 
105  uint64_t bin_dest_addr = mac_bits_for(dest_addr_);
106  if (bin_dest_addr == INVALID_MAC) {
107  std::cerr << "invalid MAC address: " << dest_addr_ << std::endl;
108  return test_afu::error;
109  }
110 
111  std::string eth_ifc = hafu->ethernet_interface();
112 
113  std::cout << "100G loopback test" << std::endl
114  << " port: " << port_ << std::endl
115  << " eth_loopback: " << eth_loopback_ << std::endl
116  << " num_packets: " << num_packets_ << std::endl
117  << " gap: " << gap_ << std::endl
118  << " src_address: " << src_addr_ << std::endl
119  << " (bits): 0x" << std::hex << bin_src_addr << std::endl
120  << " dest_address: " << dest_addr_ << std::endl
121  << " (bits): 0x" << std::hex << bin_dest_addr << std::endl
122  << " pattern: " << pattern_ << std::endl
123  << " eth: " << eth_ifc << std::endl
124  << std::endl;
125 
126  if (eth_loopback_ == "on")
127  enable_eth_loopback(eth_ifc, true);
128 
129  hafu->write64(TRAFFIC_CTRL_PORT_SEL, port_);
130 
131  uint32_t reg = num_packets_;
132  if (reg)
133  reg |= 0x80000000;
134  hafu->mbox_write(CSR_CTRL0, reg);
135 
136  hafu->mbox_write(CSR_SRC_ADDR_LO, static_cast<uint32_t>(bin_src_addr));
137  hafu->mbox_write(CSR_SRC_ADDR_HI, static_cast<uint32_t>(bin_src_addr >> 32));
138 
139  hafu->mbox_write(CSR_DST_ADDR_LO, static_cast<uint32_t>(bin_dest_addr));
140  hafu->mbox_write(CSR_DST_ADDR_HI, static_cast<uint32_t>(bin_dest_addr >> 32));
141 
142  reg = 0;
143  if (gap_ == "random")
144  reg |= 1 << 6;
145 
146  if (pattern_ == "fixed")
147  reg |= 1 << 4;
148  else if (pattern_ == "increment")
149  reg |= 2 << 4;
150 
151  hafu->mbox_write(CSR_CTRL1, reg);
152 
153  print_registers(std::cout, hafu);
154 
155  std::cout << std::endl;
156  show_eth_stats(eth_ifc);
157 
158  if (eth_loopback_ == "on")
159  enable_eth_loopback(eth_ifc, false);
160 
161  return test_afu::success;
162  }
