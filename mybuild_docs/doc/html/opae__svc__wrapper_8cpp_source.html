72 void*
73 OPAE_SVC_WRAPPER::allocBuffer(size_t nBytes, uint64_t* ioAddress)
74 {
75  fpga_result r;
76  void* va;
77 
78  //
79  // Allocate an I/O buffer shared with the FPGA. When VTP is present
80  // the FPGA-side address translation allows us to allocate multi-page,
81  // virtually contiguous buffers. When VTP is not present the
82  // accelerator must manage physical addresses on its own. In that case,
83  // the I/O buffer allocation (fpgaPrepareBuffer) is limited to
84  // allocating one page per invocation.
85  //
86 
87  if (mpfVtpIsAvailable(mpf_handle))
88  {
89  // VTP is available. Use it to get a virtually contiguous region.
90  // The region may be composed of multiple non-contiguous physical
91  // pages.
92  r = mpfVtpBufferAllocate(mpf_handle, nBytes, &va);
93  if (FPGA_OK != r) return NULL;
94 
95  if (ioAddress)
96  {
97  *ioAddress = mpfVtpGetIOAddress(mpf_handle, va);
98  }
99  }
100  else
101  {
102  // VTP is not available. Map a page without a TLB entry. nBytes
103  // must not be larger than a page.
104  uint64_t wsid;
105  r = fpgaPrepareBuffer(accel_handle, nBytes, &va, &wsid, 0);
106  if (FPGA_OK != r) return NULL;
107 
108  if (ioAddress)
109  {
110  r = fpgaGetIOAddress(accel_handle, wsid, ioAddress);
111  if (FPGA_OK != r) return NULL;
112  }
113  }
114 
115  return va;
116 }

131 fpga_result
132 OPAE_SVC_WRAPPER::findAndOpenAccel(const char* accel_uuid)
133 {
134  fpga_result r;
135 
136  // Set up a filter that will search for an accelerator
137  fpga_properties filter = NULL;
138  fpgaGetProperties(NULL, &filter);
139  fpgaPropertiesSetObjectType(filter, FPGA_ACCELERATOR);
140 
141  // Add the desired UUID to the filter
142  fpga_guid guid;
143  uuid_parse(accel_uuid, guid);
144  fpgaPropertiesSetGUID(filter, guid);
145 
146  // How many accelerators match the requested properties?
147  uint32_t max_tokens;
148  fpgaEnumerate(&filter, 1, NULL, 0, &max_tokens);
149  if (0 == max_tokens)
150  {
151  cerr << "FPGA with accelerator uuid " << accel_uuid << " not found!" << endl << endl;
152  fpgaDestroyProperties(&filter);
153  return FPGA_NOT_FOUND;
154  }
155 
156  // Now that the number of matches is known, allocate a token vector
157  // large enough to hold them.
158  fpga_token* tokens = new fpga_token[max_tokens];
159  if (NULL == tokens)
160  {
161  fpgaDestroyProperties(&filter);
162  return FPGA_NO_MEMORY;
163  }
164 
165  // Enumerate and get the tokens
166  uint32_t num_matches;
167  fpgaEnumerate(&filter, 1, tokens, max_tokens, &num_matches);
168 
169  // Not needed anymore
170  fpgaDestroyProperties(&filter);
171 
172  // Try to open a matching accelerator. fpgaOpen() will fail if the
173  // accelerator is already in use.
174  fpga_token accel_token;
175  r = FPGA_NOT_FOUND;
176  for (uint32_t i = 0; i < num_matches; i++)
177  {
178  accel_token = tokens[i];
179  r = fpgaOpen(accel_token, &accel_handle, 0);
180  // Success?
181  if (FPGA_OK == r) break;
182  }
183  if (FPGA_OK != r)
184  {
185  cerr << "No accelerator available with uuid " << accel_uuid << endl << endl;
186  goto done;
187  }
188 
189  // Map MMIO
190  fpgaMapMMIO(accel_handle, 0, NULL);
191 
192  // Connect to MPF
193  r = mpfConnect(accel_handle, 0, 0, &mpf_handle, 0);
194  if (FPGA_OK != r) goto done;
195 
196  done:
197  // Done with tokens
198  for (uint32_t i = 0; i < num_matches; i++)
199  {
200  fpgaDestroyToken(&tokens[i]);
201  }
202 
203  delete[] tokens;
204 
205  return r;
206 }

212 bool
213 OPAE_SVC_WRAPPER::probeForASE()
214 {
215  fpga_result r = FPGA_OK;
216  uint16_t device_id = 0;
217 
218  // Connect to the FPGA management engine
219  fpga_properties filter = NULL;
220  fpgaGetProperties(NULL, &filter);
221  fpgaPropertiesSetObjectType(filter, FPGA_DEVICE);
222 
223  // Connecting to one is sufficient to find ASE.
224  uint32_t num_matches = 1;
225  fpga_token fme_token;
226  fpgaEnumerate(&filter, 1, &fme_token, 1, &num_matches);
227  if (0 != num_matches)
228  {
229  // Retrieve the device ID of the FME
230  fpgaGetProperties(fme_token, &filter);
231  r = fpgaPropertiesGetDeviceID(filter, &device_id);
232  fpgaDestroyToken(&fme_token);
233  }
234  fpgaDestroyProperties(&filter);
235 
236  // ASE's device ID is 0xa5e
237  return ((FPGA_OK == r) && (0xa5e == device_id));
238 }
