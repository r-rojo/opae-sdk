34 //
35 // Manage a standard collection of CSRS. This is the software-side interface
36 // to the base cci_csrs RTL module.
37 //
38 
39 class CSR_MGR
40 {
41 protected:
42  enum {
43  //
44  // Offset of user CSRs in the CSR index space. Offsets below this
45  // are in the "common" (application-independent) CSR space,
46  // defined below.
47  //
48  // This constant must match the RTL implementation in csr_mgr.sv!
49  //
50  USER_CSR_BASE = 32
51  };
52 
53 public:
54  CSR_MGR(SVC_WRAPPER& svc) : svc(svc) {};
55 
56  ~CSR_MGR() {};
57 
58  //
59  // Write/read application-specific CSRs. The maximum CSR index
60  // is application-dependent.
61  //
62  void writeCSR(uint32_t idx, uint64_t v) {
63  svc.mmioWrite64(8 * (USER_CSR_BASE + idx), v);
64  }
65 
66  uint64_t readCSR(uint32_t idx) {
67  return svc.mmioRead64(8 * (USER_CSR_BASE + idx));
68  }
69 
70  //
71  // Common CSRs available on in all programs using CSR_MGR. These offsets
72  // must match the implementation in csr_mgr.sv.
73  //
74  typedef enum {
75  CSR_COMMON_DFH = 0,
76  CSR_COMMON_ID_L = 1,
77  CSR_COMMON_ID_H = 2,
78  // AFU frequency
79  CSR_COMMON_FREQ = 8,
80  // Number of read/write hits in the FIU system memory cache
81  CSR_COMMON_CACHE_RD_HITS = 9,
82  CSR_COMMON_CACHE_WR_HITS = 10,
83  // Lines read/written on the cached physical channel
84  CSR_COMMON_VL0_RD_LINES = 11,
85  CSR_COMMON_VL0_WR_LINES = 12,
86  // Lines read or written on the non-cached physical channels
87  CSR_COMMON_VH0_LINES = 13,
88  CSR_COMMON_VH1_LINES = 14,
89  // A collection of status signals from the FIU. See "FIU state"
90  // defined in csr_mgr.sv.
91  CSR_COMMON_FIU_STATE = 15,
92  CSR_COMMON_RD_ALMOST_FULL_CYCLES = 16,
93  CSR_COMMON_WR_ALMOST_FULL_CYCLES = 17,
94 
95  CSR_COMMON__LAST = 18
96  } t_csr_common;
97 
98  // Read one of the common CSRs
99  uint64_t readCommonCSR(t_csr_common idx) {
100  if (idx >= CSR_COMMON__LAST) return ~uint64_t(0);
101 
102  return svc.mmioRead64(8 * uint32_t(idx));
103  }
104 
105  //
106  // Return the frequency at which the AFU is running. In the common
107  // framework here, a CSR holds the frequency of the clock to which the
108  // AFU is attached. When the attached clock is the "user clock" the
109  // frequency isn't known at compile time.
110  //
111  uint64_t getAFUMHz(uint64_t uClk_usr_mhz = 0) {
112  // What's the AFU frequency (MHz)?
113  uint64_t afu_mhz = readCommonCSR(CSR_COMMON_FREQ);
114 
115  if (afu_mhz == 2) {
116  // 2 indicates uClk_usr
117  afu_mhz = uClk_usr_mhz;
118  } else if (afu_mhz == 1) {
119  // 1 indicates uClk_usrDiv2
120  afu_mhz = uClk_usr_mhz >> 1;
121  }
122  return afu_mhz;
123  }
124 
125 protected:
126  SVC_WRAPPER& svc;
127 };
