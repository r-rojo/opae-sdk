127  virtual int run(test_afu *afu, CLI::App *app) override
128  {
129  (void)app;
130 
131  hssi_afu *hafu = dynamic_cast<hssi_afu *>(afu);
132 
133  uint64_t bin_src_addr = mac_bits_for(src_addr_);
134  if (bin_src_addr == INVALID_MAC) {
135  std::cerr << "invalid MAC address: " << src_addr_ << std::endl;
136  return test_afu::error;
137  }
138 
139  uint64_t bin_dest_addr = mac_bits_for(dest_addr_);
140  if (bin_dest_addr == INVALID_MAC) {
141  std::cerr << "invalid MAC address: " << dest_addr_ << std::endl;
142  return test_afu::error;
143  }
144 
145  std::string eth_ifc = hafu->ethernet_interface();
146 
147  std::cout << "10G loopback test" << std::endl
148  << " port: " << port_ << std::endl
149  << " eth_loopback: " << eth_loopback_ << std::endl
150  << " num_packets: " << num_packets_ << std::endl
151  << " packet_length: " << packet_length_ << std::endl
152  << " src_address: " << src_addr_ << std::endl
153  << " (bits): 0x" << std::hex << bin_src_addr << std::endl
154  << " dest_address: " << dest_addr_ << std::endl
155  << " (bits): 0x" << std::hex << bin_dest_addr << std::endl
156  << " random_length: " << random_length_ << std::endl
157  << " random_payload: " << random_payload_ << std::endl
158  << " rnd_seed0: " << rnd_seed0_ << std::endl
159  << " rnd_seed1: " << rnd_seed1_ << std::endl
160  << " rnd_seed2: " << rnd_seed2_ << std::endl
161  << " eth: " << eth_ifc << std::endl
162  << std::endl;
163 
164  if (eth_loopback_ == "on")
165  enable_eth_loopback(eth_ifc, true);
166 
167  hafu->mbox_write(CSR_STOP, 0);
168 
169  hafu->write64(TRAFFIC_CTRL_PORT_SEL, port_);
170  hafu->mbox_write(CSR_MAC_LOOP, 0);
171 
172  hafu->mbox_write(CSR_NUM_PACKETS, num_packets_);
173 
174  hafu->mbox_write(CSR_PACKET_LENGTH, packet_length_);
175 
176  hafu->mbox_write(CSR_SRC_ADDR0, static_cast<uint32_t>(bin_src_addr));
177  hafu->mbox_write(CSR_SRC_ADDR1, static_cast<uint32_t>(bin_src_addr >> 32));
178 
179  hafu->mbox_write(CSR_DEST_ADDR0, static_cast<uint32_t>(bin_dest_addr));
180  hafu->mbox_write(CSR_DEST_ADDR1, static_cast<uint32_t>(bin_dest_addr >> 32));
181 
182  hafu->mbox_write(CSR_RANDOM_LENGTH, (random_length_ == "fixed") ? 0 : 1);
183  hafu->mbox_write(CSR_RANDOM_PAYLOAD, (random_payload_ == "incremental") ? 0 : 1);
184  hafu->mbox_write(CSR_RND_SEED0, rnd_seed0_);
185  hafu->mbox_write(CSR_RND_SEED1, rnd_seed1_);
186  hafu->mbox_write(CSR_RND_SEED2, rnd_seed2_);
187 
188  hafu->mbox_write(CSR_START, 1);
189 
190  print_registers(std::cout, hafu);
191 
192  uint32_t count;
193  const uint64_t interval = 100ULL;
194  do
195  {
196  count = hafu->mbox_read(CSR_PACKET_TX_COUNT);
197 
198  if (!running_) {
199  hafu->mbox_write(CSR_STOP, 1);
200  return test_afu::error;
201  }
202 
203  std::this_thread::sleep_for(std::chrono::microseconds(interval));
204  } while(count < num_packets_);
205 
206  std::cout << std::endl;
207  show_eth_stats(eth_ifc);
208 
209  if (eth_loopback_ == "on")
210  enable_eth_loopback(eth_ifc, false);
211 
212  return test_afu::success;
213  }
